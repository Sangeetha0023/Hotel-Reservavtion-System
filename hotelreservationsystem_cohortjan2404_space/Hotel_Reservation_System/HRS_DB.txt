create table residents(
RESIDENT_ID  VARCHAR(25) primary key ,
 RESIDENT_NAME  VARCHAR(25),      
AGE  	INT  ,
GENDER  VARCHAR(25)  ,    
CONTACT_NUMBER  BIGINT,
EMAIL  	VARCHAR(25),      
ADDRESS VARCHAR(25) ,     
NUMBER_OF_ADULTS  INT , 
NUMBER_OF_CHILDREN_ABOVE12  INT,
NUMBER_OF_CHILDREN_ABOVE5  	INT , 
DURATION_OF_STAY  	INT , 
RESIDENT_TYPE  	VARCHAR(25) );    
    








CREATE table riresident(

RESIDENT_ID  VARCHAR(25) ,
   
IDPROOF_AADHARNO  	BIGINT,
CONSTRAINT  FOREIGN KEY(RESIDENT_ID) REFERENCES residents(RESIDENT_ID));

create table nriresident(
RESIDENT_ID VARCHAR(25) primary key,  
PASSPORT_NO 	VARCHAR(26) ,    
PASSPORT_TYPE 	VARCHAR(27) ,    
NATIONALITY 	VARCHAR(28) ,    
PURPOSEFORVISIT 	VARCHAR(29),
CONSTRAINT  FOREIGN KEY(RESIDENT_ID) REFERENCES residents(RESIDENT_ID));

create table room(
ROOM_NUMBER 	VARCHAR(25)  primary key,  
FLOOR_NUMBER 	VARCHAR(25) ,    
IS_OCCUPIED 	VARCHAR(25),     
MAX_OCCUPANCY 	INT,
HAS_AC 	VARCHAR(25),     
HAS_SWIMMING_POOL_ACCESS 	VARCHAR(25),     
HAS_GYM_ACCESS 	VARCHAR(25));    

create table booking (
BOOKING_ID  	VARCHAR(26)  Primary Key,  
RESIDENT_ID  	VARCHAR(26) , 
RESIDENT_NAME  	VARCHAR(26),
DURATION_OF_STAY  	INT,  

CHECK_IN_DATE  	DATE,  
CHECK_OUT_DATE  	DATE , 
FLOOR_NUMBER  	VARCHAR(25),      
ROOM_NUMBER  	VARCHAR(26),      
PREFERRED_PACKAGE  	VARCHAR(27) ,     
AC_ACCESS  	VARCHAR(28) ,     
POOL_ACCESS  	VARCHAR(29),      
GYM_ACCESS  	VARCHAR(30),
 CONSTRAINT  FOREIGN KEY(RESIDENT_ID) REFERENCES residents(RESIDENT_ID)
   
);     
create table payment
(
    PAYMENT_ID 	VARCHAR(26) Primary Key, 
BOOKING_ID 	VARCHAR(26) ,
RESIDENT_NAME 	VARCHAR(26),     
FLOOR_NUMBER 	VARCHAR(26) ,    
ROOM_NUMBER 	VARCHAR(26) ,    
CHECK_IN_DATE 	DATE, 
CHECK_OUT_DATE 	DATE, 
PAYMENT_DATE 	DATE, 
PAYMENT_METHOD 	VARCHAR(26) ,    
BILL_AMOUNT 	DOUBLE,
CONSTRAINT Foreign Key(BOOKING_ID) REFERENCES booking(BOOKING_ID));
